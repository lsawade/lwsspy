class DummyFile(object):
    def write(self, x): pass



[docs]@contextlib.contextmanager
def nostdout():
    """Context manager that suppresses stdout of a function.
    """
    save_stdout = sys.stdout
    sys.stdout = DummyFile()
    yield
    sys.stdout = save_stdout
